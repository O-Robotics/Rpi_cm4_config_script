[all]
kernel=vmlinuz
cmdline=cmdline.txt
initramfs initrd.img followkernel

# Enable the audio output, I2C and SPI interfaces on the GPIO header
dtparam=audio=on
dtparam=i2c_arm=on
dtparam=spi=on

# Enable 1-Wire
dtoverlay=w1-gpio

# Enable UART for RS485
dtoverlay=uart3
dtoverlay=uart5

# Enable SPI and configure CAN bus
dtoverlay=mcp2515-can0,oscillator=16000000,interrupt=25

# Enable Ethernet for CM5 (CM4 used different overlays)
dtoverlay=rpivid-v4l2
dtoverlay=mdio-gpio  # Ensures MDIO is enabled for Ethernet
dtoverlay=dwc2,dr_mode=host  # USB controller for CM5

# Ethernet PHY reset (CM5 uses updated networking settings)
dtparam=eth_led0=14
dtparam=eth_led1=15

# Disable overscan to avoid display issues
disable_overscan=1

# Enable full KMS graphics and fix memory allocation issues
dtoverlay=vc4-kms-v3d,cma-128
disable_fw_kms_setup=1

# Enable PCIe for NVMe SSD (Optional, but necessary if using NVMe)
dtparam=pciex1

[cm5]
# Enable USB2 outputs on the IO board (CM5 equivalent of CM4 setting)
dtoverlay=dwc2,dr_mode=host
dtparam=usb=on

[all]
# Auto-detect connected cameras and displays (for libcamera support)
camera_auto_detect=1
display_auto_detect=1

# Enable 64-bit mode
arm_64bit=1
